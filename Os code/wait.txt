#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>

int main() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("fork failed");
        exit(EXIT_FAILURE);
    }
    else if (pid == 0) {   // Child process
        printf("Child: PID %d\n", getpid());
        sleep(2);
        printf("Child: Exiting\n");
        exit(42);          // Child exits with custom code 42
    }
    else {                 // Parent process
        int status;
        pid_t wpid = wait(&status);  // Parent waits for child to finish
        printf("Parent: Child with PID %d exited with status %d\n",
               wpid, WEXITSTATUS(status));
    }

    return 0;
}
